const get = (from, ...selectors) =>
    [...selectors]
        .filter(s => typeof s === 'string')
        .map(s =>
            s
                .replace(/\[([^\[\]]*)\]/g, '.$1.')
                .split('.')
                .filter(t => t !== '')
                .reduce((prev, cur) => prev && prev[cur], from)
        );
