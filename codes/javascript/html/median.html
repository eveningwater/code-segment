const median = (arr) => {
  const mid = Math.floor(arr.length / 2),
    newArr = arr.sort((a, b) => a - b);
  return arr.length % 2 !== 0
    ? newArr[mid]
    : (newArr[mid - 1] + newArr[mid]) / 2;
};
